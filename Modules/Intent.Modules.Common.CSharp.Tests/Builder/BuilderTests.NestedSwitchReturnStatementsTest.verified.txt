using System;

[assembly: DefaultIntentManaged(Mode.Fully)]

namespace Namespace
{
    public class Class
    {
        public string SwitchReturnStatements(IEnumerable<string> collection)
        {
            foreach (var item in collection)
            {
                switch (item)
                {
                    case "Item1":
                        return "Item1";
                    case "Item2":
                        return "Item2";
                }
            }
            return "Item X";
        }
    }
}
